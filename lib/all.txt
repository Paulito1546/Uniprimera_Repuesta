  final url = Uri.parse('http://10.0.2.2:3000/sendmail');
  final response = await http.post(
    url,
    headers: {'Content-Type': 'application/json'},
    body: jsonEncode({
      'subject': 'Alerte urgence',
      'text': jsonEncode(report),
    }),
  );
  if (response.statusCode != 200) {
    throw Exception('Ã‰chec envoi rapport');
  }

class _QuestionnairePageState extends State<QuestionnairePage> {
  bool? isSelfEmergency; // SÃ­: true, No: false
  bool? breathes; // For "Â¿Respira?"
  bool? unconscious; // For "Â¿EstÃ¡ inconsciente?"
  bool? coherent; // For "Â¿Responde de manera coherente?"
  final TextEditingController _descriptionController = TextEditingController();
  bool showDescription = false;
  bool showAdditionalQuestions = false;
  bool stoppedEarly = false;

  Future<Map<String, dynamic>> _buildReport() async {
    final prefs = await SharedPreferences.getInstance();
    final firstName = prefs.getString('first_name') ?? 'Inconnu';
    final lastName = prefs.getString('last_name') ?? 'Inconnu';
    final idNumber = prefs.getString('id_number') ?? 'Inconnu';

    String location = 'X';
    try {
      bool serviceEnabled = await Geolocator.isLocationServiceEnabled();
      if (!serviceEnabled) throw Exception('Service localisation dÃ©sactivÃ©');

      LocationPermission permission = await Geolocator.checkPermission();
      if (permission == LocationPermission.denied) {
        permission = await Geolocator.requestPermission();
        if (permission == LocationPermission.denied) throw Exception('Permission refusÃ©e');
      }

      if (permission == LocationPermission.deniedForever) throw Exception('Permission refusÃ©e dÃ©finitivement');

      Position position = await Geolocator.getCurrentPosition(desiredAccuracy: LocationAccuracy.high);
      location = 'lat:${position.latitude}, long:${position.longitude}';
    } catch (e) {
      ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text('Erreur localisation: $e')));
    }

    return {
      'Ponente': '$firstName $lastName',
      'Identidad': idNumber,
      'Persona en situaciÃ³n de emergencia': isSelfEmergency == true ? 'yo' : 'otro',
      'Â¿Respira?': breathes == null ? 'X' : (breathes! ? 'Si' : 'No'),
      'Â¿EstÃ¡ inconsciente?': unconscious == null ? 'X' : (unconscious! ? 'Si' : 'No'),
      'Â¿Responde de manera coherente a preguntas sencillas?': coherent == null ? 'X' : (coherent! ? 'Si' : 'No'),
      'MÃ¡s': _descriptionController.text.isEmpty ? 'X' : _descriptionController.text,
      'Localisation': location,
      'Type': widget.emergencyType, // Bonus: AjoutÃ© pour contexte
    };
  }

  void _handleFirstQuestion(bool answer) {
    setState(() {
      isSelfEmergency = answer;
      if (answer) {
        showDescription = true; // Passer directement Ã  description
      } else {
        showAdditionalQuestions = true; // Afficher questions supplÃ©mentaires
      }
    });
  }

  void _handleAdditionalQuestion(String question, bool answer) {
    setState(() {
      if (question == 'Â¿Respira?') breathes = answer;
      if (question == 'Â¿EstÃ¡ inconsciente?') unconscious = answer;
      if (question == 'Â¿Responde de manera coherente a preguntas sencillas?') coherent = answer;

      if (answer) { // Si "SÃ­" Ã  n'importe laquelle, arrÃªter et passer Ã  description
        stoppedEarly = true;
        showDescription = true;
      }
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Questionnaire')),
      body: SingleChildScrollView(
        padding: const EdgeInsets.all(16.0),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [

                ],
              ),
            ],
            if (showAdditionalQuestions && breathes != null && unconscious == null && !stoppedEarly) ...[
              const SizedBox(height: 24),
              const Text('Â¿EstÃ¡ inconsciente?', style: TextStyle(fontSize: 18)),
              const SizedBox(height: 8),
              Row(
                children: [
                  ElevatedButton(onPressed: () => _handleAdditionalQuestion('Â¿EstÃ¡ inconsciente?', true), child: const Text('SÃ­')),
                  const SizedBox(width: 16),
                  ElevatedButton(onPressed: () => _handleAdditionalQuestion('Â¿EstÃ¡ inconsciente?', false), child: const Text('No')),
                ],
              ),
            ],
            if (showAdditionalQuestions && unconscious != null && coherent == null && !stoppedEarly) ...[
              const SizedBox(height: 24),
              const Text('Â¿Responde de manera coherente a preguntas sencillas?', style: TextStyle(fontSize: 18)),
